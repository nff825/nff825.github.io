能量有限元(MATLAB)

2025/4/30 能量有限元 MATLAB EFEA

主程序

功能

指定参数空间（激励的位置、幅值、频率），进行能量有限元分析

代码

%% 获取数据 不同激励设置
clear; close all; clc
%% ==========创建数据存储文件夹 ==========
output_folder = 'EFEA_data_20250429_excitation_1800';
if ~exist(output_folder, 'dir')
    mkdir(output_folder);
end
%% ========== 设置固定参数 ==========
geometry.length_x = 0.5075;    % 薄板在 x 方向的长度
geometry.length_y = 0.4375;    % 薄板在 y 方向的长度
geometry.h = 42.4382e-3;       % 薄板的厚度

num_elements_x = 30;            % x 方向的单元数量
num_elements_y = 30;            % y 方向的单元数量
num_nodes = (num_elements_x+1)^2; % 节点数量

material.E = 1.6966e9;          % 弹性模量
material.rho = 94.2547;         % 密度
material.mu = 0.3;              % 泊松比
material.damping_ratio = 0.05; % 阻尼比
%% ========== 设置可变参数范围 ==========
frequencies = [500:10:550];
amplitudes = [1:0.1:1.5];
excitation_shifts = generate_excitation_shifts(num_nodes, 50);
COUNT = numel(frequencies) * numel(amplitudes) * numel(excitation_shifts);
%% ========== 生成数据 (并行计算) ==========
delete(gcp); % 关闭并行池
CoreNum = 4; % 设定机器CPU核心数量
parpool(CoreNum); % 开启并行池
parfor counter = 1:COUNT
    [frequency, F, offset] = get_parameters(counter, frequencies, amplitudes, excitation_shifts);

    %% ========== 激励设置 ==========
    excitation_freq = frequency;                    % 激励频率
    excitation_amp = F;                              % 激励幅值
    excitation_pos = round(num_nodes/2) + offset;    % 激励位置
    %% ========== 能量有限元分析 ==========
    output_filename = sprintf('%s/data_%d.txt', output_folder, counter);
    energy_density = EFEM_2D_plate(geometry, material, struct('freq', excitation_freq, 'amp', excitation_amp, 'pos', excitation_pos), num_elements_x, num_elements_y);
    %% ========== 写入参数数据 ==========
    fileID = fopen(output_filename, 'w');
    fprintf(fileID, '%.6f %.6f %.6f %.6f %.6f %.6f %.6f %.6f %.6f %.6f\n', ...
        excitation_freq, excitation_amp, excitation_pos, ...
        material.E, material.rho, material.mu, material.damping_ratio, ...
        geometry.length_x, geometry.length_y, geometry.h);
    %% ========== 换算能量密度（dB） ==========
    E = reshape(energy_density, num_elements_x+1, num_elements_y+1);
    E_ref = 1e-12;
    E(E < E_ref) = E_ref;
    E_dB = 20 * log10(E / E_ref);
    %% ========== 写入能量密度数据 ==========
    [x_coords, y_coords] = meshgrid(linspace(0, geometry.length_x, num_elements_x+1), linspace(0, geometry.length_y, num_elements_y+1));
    for i = 1:numel(x_coords)
        fprintf(fileID, '%.6f %.6f %.6f\n', x_coords(i), y_coords(i), E_dB(i));
    end
    fclose(fileID);
    fprintf('%d/%d | freq = %d amp = %d offset = %d | %s\n', counter, COUNT, frequency, F, offset, output_filename);
end

disp('done');
